### 2. "Thinkful - List and Loop Drills: Lists of lists"

function processData(data) { <br>
    let add = [] <br>
    let out = 1 <br>
    for (let x of data) { <br>
        add.push(Number(x[0]-Number(x[1])))  <br>
    } <br>
    for (let y of add) { <br>
        out = out * Number(y)   <br>
    } <br>
    return out <br>
} <br>
